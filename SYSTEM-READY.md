    query(prompt, model) {
        // Route intelligently
        if (modelTrainer.has(model)) {
            return trainedModel.generate(prompt);
        } else if (ollama.available()) {
            return ollama.query(model, prompt);
        } else {
            return algorithmicGen.generate(prompt);
        }
    }
